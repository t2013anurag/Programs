void insertion_sort(int x[],int length)
{
  int key,i;
  for(int j=1;j<length;j++)
  {
     key=x[j];
     i=j-1;
     while(x[i]>key && i>=0)
     {
	       x[i+1]=x[i];
	 i--;
     }
     x[i+1]=key;
  }
}
